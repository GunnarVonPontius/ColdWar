# Coca Cola informs US DoS of the French ban
country_event = {
	id = fra.500
	title = "Coca Cola informs the DoS of a French ban on their products"

		tag = USA

		name = "The French must respect trade regulations!"

			factor = 60
		}
		hidden_effect = {
			FRA = { country_event = { days = 5 id = fra.501 } }
		}
	}
}
