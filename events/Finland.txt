add_namespace = finland

#PAASIKIVI VS. PEKKALA VS. KEKKONEN

country_event = {
	id = finland.501701
	title = finland.501701.t 
	desc = finland.501701.d 
	
	picture = GFX_FIN_election
		
	trigger = {
		tag = FIN
		date > 1950.1.1
		date < 1951.1.1
	}
	
	fire_only_once = yes
	
	#PAASIKIVI WINS 
	
	option = {
		name = finland.501701.a 
		set_country_flag = PAASIKIVI_WINS
	}
	
	#PEKKALA WINS 
	option = {
		name = finland.501701.b
		
		retire_country_leader = yes 
		
		set_party_name = { ideology = conservatism long_name = FIN_peoples_democratic_league_long name = FIN_peoples_democratic_league  }
		
		create_country_leader = {
			name = "Mauno Pekkala"
			desc = "POLITICS_MICHAEL_JOSEPH_SAVAGE_DESC"
			picture = "Portrait_Finland_Mauno_Pekkala.dds"
			expire = 2000.1.1 
			ideology = conservative
		}
		
		set_country_flag = PEKKALA_WINS 
	}
	#KEKKONEN WINS 
	option = {
		name = finland.501701.c 
	
		retire_country_leader = yes 
	
		set_party_name = { ideology = liberalism long_name = FIN_agrarian_party_long name = FIN_agrarian_party}
	
		create_country_leader = {
			name = "Urho Kekkonen"
			desc = "POLITICS_MICHAEL_JOSEPH_SAVAGE_DESC"
			picture = "Portrait_Finland_Urho_Kekkonen.dds"
			expire = 2000.1.1
			ideology = conservative
		}
		
		set_country_flag = KEKKONEN_WINS 
	}
}

#KEKKONEN VS FAGERHOLM (1956 election historical)

		}

	option = {
