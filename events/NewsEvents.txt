# 1952 Egyptian Revolution

	id = news.egyrevolution
	title = news.egyrevolution.t
	picture = 
	
	major = yes
	
	is_triggered_only = yes

	immediate = {
		EGY = {
			set_cosmetic_tag = EGY_REP
			create_country_leader = {
				name = "Mohammed Naguib"
				desc = "POLITICS_NAGUIB_DESC"
				picture = "Portrait_Egypt_Mohammed_Naguib.dds"
				expire = "1984.8.28"
				ideology = despotic
			}
			set_party_name = { ideology = neutrality long_name = EGY_free_officers_movement name = EGY_free_officers }
		}
	}
	
	desc = {
		text = news.egyrevolution.d_egy
		trigger = {
			tag = EGY
		}
	}
	
	desc = {
		text = news.egyrevolution.d_row
		trigger = {
			NOT = { tag = EGY }
		}
	}

			tag = EGY
		}
	}

		name = news.egyrevolution.b
		trigger = {

		}
	}
